	public void adivinhar(char c) { 
		 
		if (!Character.isLetter(c)){
			txtKk.setText("APENAS LETRAS!");
			return; 
		}else{
			
			txtKk.setText("");
			textField_3.setText("");
		}
 
		if (PALAVRA_CECRETA.indexOf(c) == -1){
			ERROS_COMETIDOS++;
			chances = ERROS_PERMITIDOS - ERROS_COMETIDOS;
			txtKk.setText("palpite errado , tente novamente");
			textField_3.setText("suas chances restantes são "+chances);
			
			switch (ERROS_COMETIDOS) {
			case 1:
				txtO.setText("O");
			break;
			case 2:
				txtL.setText("l");
			break;
			case 3:
				textField_2.setText("/");
			break;
			case 4:
				textField_4.setText("/");
			break;
			case 5:
				textField_5.setText("\\");
			break;
			case 6:
				textField_6.setText("\\");
			break;
			default:
			System.out.println("Perdeu");
			}
				
	
			
			if (ERROS_COMETIDOS==ERROS_PERMITIDOS){
				
				JOptionPane.showMessageDialog(null,"* * * * * * * VOCÊ PERDEU * * * * * *","Ganhador", JOptionPane.INFORMATION_MESSAGE);
				textField_1.setText("A PALAVRA ESCONDIDA É: "+PALAVRA_CECRETA);
				cont=8;
			}
		} else {
			txtKk.setText("CORRETO!");
			CONTADOR_LETRAS = CONTADOR_LETRAS + c;
		 int nu = 0;
		 String letra ="";
			for (int i = 0; i < PALAVRA_CECRETA.length(); i++) {
				char ch = PALAVRA_CECRETA.charAt(i);
				if (CONTADOR_LETRAS.contains(String.valueOf(ch))) {
					letra = letra + ch;
					nu++;
				} else {
					letra = letra + "-";
				}
			}
			textField_1.setText(letra);
			System.out.println();
	        if (nu == PALAVRA_CECRETA.length()){
	        	ganhou();
	        	cont = 8;
	        }
		}

	}
